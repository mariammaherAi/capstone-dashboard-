📋 Setup & Connection Guide

01

Wire the Arduino

Turbidity sensor AOUT → A0 · LM35 VOUT → A1 · Relay IN → Pin 7. Power with 5 V.

02

Upload the Sketch

Paste the code below into Arduino IDE. Select Arduino Uno board + correct COM port, then Upload.

03

Use Chrome or Edge

Web Serial API requires Google Chrome or Microsoft Edge v89+. Open this HTML file directly.

04

Click Connect Arduino

Press the header button. Select your Arduino COM port from the browser dialog. Live data streams instantly.

05

Verify Data Format

Arduino must send exactly:
Turbidity: 350 | Temp: 26.50 Pump ON

06

Pump Logic (Arduino-side)

Pump ON when turbidity > 300 OR temp < 20 OR temp > 30 °C. Thresholds editable in sketch.

⚡ ARDUINO SKETCH — hydrosense.ino 
// ╔══════════════════════════════════════════════════════════════╗
// ║  HydroSense — Smart Aquaculture Arduino Uno Sketch          ║
// ║  Pins    : A0 = Turbidity, A1 = LM35, D7 = Relay           ║
// ║  Output  : "Turbidity: XXX | Temp: XX.XX Pump ON/OFF"       ║
// ║  Interval: 2 seconds  |  Baud: 9600                        ║
// ╚══════════════════════════════════════════════════════════════╝

// ─── Pin Definitions ────────────────────────────────────────────
const int TURBIDITY_PIN = A0;  // Turbidity sensor analog output
const int TEMP_PIN      = A1;  // LM35 temperature sensor analog output
const int RELAY_PIN     = 7;   // Relay IN pin  (HIGH = pump ON)

// ─── Control Thresholds ─────────────────────────────────────────
const int   TURB_LIMIT  = 300;   // Turbidity raw value above this triggers pump
const float TEMP_MIN    = 20.0;  // Below this temperature → pump ON
const float TEMP_MAX    = 30.0;  // Above this temperature → pump ON

// ─── Timing ─────────────────────────────────────────────────────
unsigned long prevMillis = 0;
const long    INTERVAL   = 2000; // Transmit every 2000 ms

// ────────────────────────────────────────────────────────────────
void setup() {
  Serial.begin(9600);                      // Must match dashboard baud rate
  pinMode(RELAY_PIN, OUTPUT);             // Relay pin as digital output
  digitalWrite(RELAY_PIN, LOW);          // Ensure pump is OFF at startup
  Serial.println("HydroSense v2.0 Ready"); // Boot message
}

// ────────────────────────────────────────────────────────────────
void loop() {
  if (millis() - prevMillis >= INTERVAL) {
    prevMillis = millis();

    // ── 1. Read Turbidity ──────────────────────────────────────
    //    Raw ADC 0–1023. Higher raw → cleaner water (more light).
    //    We map so higher output value = dirtier water (0–1000 scale).
    int rawTurb   = analogRead(TURBIDITY_PIN);
    int turbidity = map(rawTurb, 0, 1023, 1000, 0);
    turbidity     = constrain(turbidity, 0, 1000);

    // ── 2. Read Temperature via LM35 ───────────────────────────
    //    LM35 outputs 10 mV per °C.
    //    Voltage (mV) = (rawADC / 1023.0) × 5000
    //    Temperature  = voltage / 10.0
    int   rawTemp = analogRead(TEMP_PIN);
    float voltage = (rawTemp / 1023.0) * 5000.0;  // in mV
    float tempC   = voltage / 10.0;               // LM35: 10 mV / °C

    // ── 3. Pump Control Logic ──────────────────────────────────
    //    Activate pump if any condition is out of range
    bool pumpOn = (turbidity > TURB_LIMIT) ||
                   (tempC    < TEMP_MIN)    ||
                   (tempC    > TEMP_MAX);

    digitalWrite(RELAY_PIN, pumpOn ? HIGH : LOW);

    // ── 4. Send Serial Data in EXACT required format ───────────
    //    Format: "Turbidity: 350 | Temp: 26.50 Pump ON"
    Serial.print("Turbidity: ");
    Serial.print(turbidity);
    Serial.print(" | Temp: ");
    Serial.print(tempC, 2);          // 2 decimal places
    Serial.print(" Pump ");
    Serial.println(pumpOn ? "ON" : "OFF");
  }
}

/* ─── Wiring Reference ─────────────────────────────────────────
   Turbidity Sensor (SEN0189 or analog type):
     VCC  →  5V      GND  →  GND      AOUT →  A0

   LM35 Temperature Sensor (TO-92 package):
     +VS  →  5V      GND  →  GND      VOUT →  A1

   5V Relay Module:
     VCC  →  5V      GND  →  GND      IN   →  Pin 7
     COM  →  Pump power line
     NO   →  Pump positive lead (Normally Open contact)
   ─────────────────────────────────────────────────────────── */
